* Productos.
CREATE TABLE tn_prod ( ;
    codigo N(7), ;
    tipo N(2), ;
    nombre V(60), ;
    descripcio M(4), ;
    descr_cort V(200), ;
    sku V(50), ;
    precio N(12), ;
    cantidad N(9,2), ;
    visibilida N(2), ;
    impuesto L(1), ;
    peso N(10,4), ;
    created_at T(10), ;
    updated_at T(10) NULL ;
)
INDEX ON codigo TAG 'indice1' CANDIDATE FOR !DELETED()
INDEX ON nombre TAG 'indice2' CANDIDATE FOR !DELETED()
INDEX ON sku TAG 'indice3' CANDIDATE FOR !DELETED()

* Precio especial (descuento por tiempo).
CREATE TABLE tn_proes ( ;
    producto N(7), ;
    precio N(12), ;
    desde D(8), ;
    hasta D(8), ;
    created_at T(10), ;
    updated_at T(10) NULL ;
)
INDEX ON STR(producto, 7) + DTOS(desde) + DTOS(hasta) TAG 'indice1' CANDIDATE FOR !DELETED()

* Visibilidad de productos.
CREATE TABLE tn_visib ( ;
    codigo N(2), ;

    updated_at T(10) NULL ;
)
INDEX ON codigo TAG 'indice1' CANDIDATE FOR !DELETED()
INDEX ON nombre TAG 'indice2' CANDIDATE FOR !DELETED()
